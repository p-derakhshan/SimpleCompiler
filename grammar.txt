ZERO -> START EOF
START -> BEGIN MAIN END  
MAIN -> STATEMENT MAIN
MAIN -> ~    
STATEMENT -> LOOP MAIN UNTIL LP BOOLEXP RP 
STATEMENT -> CONDITION LP BOOLEXP RP BLOCK MAIN BLOCK
STATEMENT -> SL_COMMENT
STATEMENT -> ML_COMMENT   
STATEMENT -> TYPE ASSIGN
STATEMENT -> ASSIGN
ASSIGN -> ID VARTAIL ASGN EXP EOL 
TYPE -> STR   
TYPE -> FLT     
TYPE -> NUM
VARTAIL -> LP VAR_INT RP   
VARTAIL -> ~       
EXP -> PRIOR2 TAIL2
TAIL2 -> ADD PRIOR2 TAIL2   
TAIL2 -> SUB PRIOR2 TAIL2   
TAIL2 -> ~   
PRIOR2 -> TERM TAIL1    
TAIL1 -> MUL TERM TAIL1   
TAIL1 -> DIV TERM TAIL1   
TAIL1 -> REMAINDER TERM TAIL1   
TAIL1 -> ~ 
TERM -> LP EXP RP   
TERM -> ID     
TERM -> VAR_INT     
TERM -> VAR_FLOAT
TERM -> DQUOTE VAR_STRING DQUOTE
TERM -> ARRAY LB VAR_INT RB
BOOLEXP -> TERM BOOLTAIL
BOOLTAIL -> GREATER_EQL TERM  
BOOLTAIL -> SMALLER_EQL TERM    
BOOLTAIL -> GREATER TERM    
BOOLTAIL -> SMALLER TERM    
BOOLTAIL -> EQL TERM
